const defaultSize = 1024 * 1024 * 100 // 默认每一个分片的大小为100MB

function createChunks (file) {
  let current = 0
  let parts = []
  while(current < file.size) {
    const chunk = file.slice(current, current + defaultSize)
    parts.push({
      chunk,
      size: chunk.size
    })
    current += defaultSize
  }
  return parts
}
